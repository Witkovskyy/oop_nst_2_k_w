81static void undoMove(Board& board, const Move& move, const Undo& undo)
82{
83 board.squares[move.from.row][move.from.col] = undo.pieceMoved;
84 board.squares[move.to.row][move.to.col] = undo.pieceCaptured;
85
86 if (undo.pieceMoved) {
87 undo.pieceMoved->setPosition(move.from);
88 }
89 // Double check for captured piece
90 if (undo.pieceCaptured) {
91 undo.pieceCaptured->setPosition(move.to);
92 }
93 // Handle promotion revert
94 if (move.promotion != 0) {
95 undo.pieceMoved->setSymbol('P');
96 }
97
98 int fromSq = move.from.row * 8 + move.from.col;
99 int toSq = move.to.row * 8 + move.to.col;
100 Piece* p = undo.pieceMoved; // Piece that moved
101 int pIdx = getPieceIndex(p->getSymbol(), p->getColor());
102
103 // Revert side
104 board.zobristKey ^= sideKey;
105
106 // Bring back the Zobrist key
107 if (move.promotion != 0) {
108 int promoIdx = getPieceIndex(move.promotion, p->getColor());
109 board.zobristKey ^= pieceKeys[promoIdx][toSq];
110 }
111 else {
112 board.zobristKey ^= pieceKeys[pIdx][toSq];
113 }
114
115 // Revert captured
116 if (undo.pieceCaptured) {
117 int capIdx = getPieceIndex(undo.pieceCaptured->getSymbol(), undo.pieceCaptured->getColor());
118 board.zobristKey ^= pieceKeys[capIdx][toSq];
119 }
120
121 // Insert piece back

145{
146 undo.from = move.from;
147 undo.to = move.to;

150 undo.promotion = move.promotion;

153 int toSq = move.to.row * 8 + move.to.col;
154
155 Piece* p = board.getPieceAt(move.from);
156 int pIdx = getPieceIndex(p->getSymbol(), p->getColor());
157
158 // Zobrist remove piece
159 board.zobristKey ^= pieceKeys[pIdx][fromSq];
160 // Zobrist If captured, remove captured piece
161 if (undo.pieceCaptured) {
162 int capIdx = getPieceIndex(undo.pieceCaptured->getSymbol(), undo.pieceCaptured->getColor());
163 board.zobristKey ^= pieceKeys[capIdx][toSq];
164 }
165
166 // Zobrist insert piece
167 // If promoted put new figure
168 if (move.promotion != 0) {
169 int promoIdx = getPieceIndex(move.promotion, p->getColor());
170 board.zobristKey ^= pieceKeys[promoIdx][toSq];
171 }
172 else {
173 board.zobristKey ^= pieceKeys[pIdx][toSq];
174 }
175
176 // Switch side
177 board.zobristKey ^= sideKey;
178
179 // Raw move application
180 // Delete piece
181 board.squares[move.from.row][move.from.col] = nullptr;
182 // Place piece
183 board.squares[move.to.row][move.to.col] = undo.pieceMoved;
184
185 // Update piece position
186 if (undo.pieceMoved) {
187 undo.pieceMoved->setPosition(move.to);
188 }
189 // Handle promotion
190 if (move.promotion != 0) {
191 undo.pieceMoved->setSymbol(move.promotion);
192 }
193}

363bool isInCheck(Board& board, int color)
364{
365 // Find king position - robust search
366 Position kingPos = { -1, -1 };
367 for (int r = 0; r < 8; ++r) {
368 for (int c = 0; c < 8; ++c) {
369 Piece* p = board.squares[r][c];
370 if (p && p->getColor() == color && toupper(p->getSymbol()) == 'K') {
371 kingPos = { r, c };
372 goto found_king;
373 }
374 }
375 }
376found_king:;
377
378 // No king found, we treat as game over (or checkmate in specific context)
379 if (kingPos.row == -1) return true;
380
381 // Check king square
382 int enemyColor = (color == 0) ? 1 : 0;
383 return isSquareAttacked(board, kingPos, enemyColor);
384}

395int eval(const Board& board, int color)
396{
397 int whiteScore = 0;
398 int blackScore = 0;
399
400 for (int row = 0; row < 8; ++row)
401 {
402 for (int col = 0; col < 8; ++col)
403 {
404 const Piece* cell = board.squares[row][col];
405 if (!cell || cell->symbol == 'E') continue;
406
407 int material = pieceValFromSymbol(cell->symbol);
408 // PST (Piece Square Table)
409 int positional = 0;
410 char sym = toupper(cell->symbol);
411 switch (sym) {
412 case 'P': positional = getPstValue(pawnPST, row, col, cell->color); break;
413 case 'N': positional = getPstValue(knightPST, row, col, cell->color); break;
414 case 'B': positional = getPstValue(bishopPST, row, col, cell->color); break;
415 case 'R': positional = getPstValue(rookPST, row, col, cell->color); break;
416 case 'Q': positional = getPstValue(queenPST, row, col, cell->color); break;
417 case 'K': positional = getPstValue(kingPST, row, col, cell->color); break;
418 }
419
420 int total = material + positional;
421 if (cell->color == 0) // White
422 whiteScore += total;
423 else // Black
424 blackScore += total;
425 }
426 }
427
428 int finalScore = whiteScore - blackScore;
429 return finalScore * color;
430}

485static int scoreMove(const Move& move) {
486 // CAPTURES (MVV-LVA)
487 if (move.pieceCaptured != nullptr) {
488 int victimVal = pieceValFromSymbol(move.pieceCaptured->getSymbol());
489 int attackerVal = pieceValFromSymbol(move.pieceMoved->getSymbol());
490
491 // Equation: Victim * 10 - Attacker.
492 return 10000 + (victimVal * 10) - attackerVal;
493 }
494
495 // PROMOTION BONUS
496 char sym = toupper(move.pieceMoved->getSymbol());
497 if (sym == 'P') {
498 int r = move.to.row;
499 // IF PAWN LAST RANK
500 if (r == 0 || r == 7) {
501 return 9000;
502 }
503 }
504
505 return 0;
506}

571int quiescence(Board& board, int alpha, int beta, int color)
572{
573 int stand = eval(board, color);
574 if (stand >= beta)
575 return beta;
576 if (stand > alpha)
577 alpha = stand;
578
579 auto caps = generateCaptures(board, to01(color));
580 // Only consider capture moves
581 orderMoves(caps);
582 for (auto& move : caps)
583 {
584 Undo undo;
585 applyMove(board, move, undo);
586
587 int score = -quiescence(board, -beta, -alpha, -color);
588 undoMove(board, move, undo);
589 if (score >= beta)
590 return beta;
591 if (score > alpha)
592 alpha = score;
593 }
594 return alpha;
595}

639int negamax(Board& board, int depth, int alpha, int beta, int color)
640{
641 if (isRepetition(board)) {
642 // 0 is equal position, slight minus for engine to avoid repetition
643 return -35;
644 }
645
646 ++nodesVisited;
647 int ttScore;
648 Move ttMove;
649 // Read hash
650 unsigned long long key = board.zobristKey;
651 if (TT.probe(key, depth, alpha, beta, ttScore, ttMove)) {
652 return ttScore;
653 // Score found in TT, immediately return
654 }
655
656 if (depth == 0 || gameOver(board))
657 return quiescence(board, alpha, beta, color);
658
659 auto moves = legalMoves(board, to01(color));
660
661 if (moves.empty()) {
662 // No legal moves, check for checkmate or stalemate
663 if (isInCheck(board, to01(color))) return -100000 + depth; // CHECKMATE
664 return 0; // STALEMATE
665 }
666
667 orderMoves(moves);
668
669 int best = -INF;
670 Move bestMove;
671 // For TT storage
672 int oldAlpha = alpha;
673 for (auto& move : moves)
674 {
675 Undo undo;
676 applyMove(board, move, undo);
677 int score = -negamax(board, depth - 1, -beta, -alpha, -color);
678 undoMove(board, move, undo);
679 if (score > best)
680 best = score;
681 if (score > alpha)
682 alpha = score;
683 if (alpha >= beta)
684 break;
685 // beta cutoff
686 }
687
688 TTFlag flag = TT_EXACT;
689 if (best <= oldAlpha) flag = TT_ALPHA; // Not better than alpha
690 else if (best >= beta) flag = TT_BETA;
691 // Cutoff
692
693 TT.store(key, best, depth, flag, bestMove);
694
695 return best;
696}
