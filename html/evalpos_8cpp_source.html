61static void undoMove(Board& board, const Move& move, const Undo& undo)
62{
63 board.squares[move.from.row][move.from.col] = undo.pieceMoved;
64 board.squares[move.to.row][move.to.col] = undo.pieceCaptured;
65
66 if (undo.pieceMoved) {
67 undo.pieceMoved->setPosition(move.from);
68 }
69 // Double check for captured piece
70 if (undo.pieceCaptured) {
71 undo.pieceCaptured->setPosition(move.to);
72 }
73 // Handle promotion revert
74 if (move.promotion != 0) {
75 undo.pieceMoved->setSymbol('P');
76 }
77
78 int fromSq = move.from.row * 8 + move.from.col;
79 int toSq = move.to.row * 8 + move.to.col;
80 Piece* p = undo.pieceMoved; // Piece that moved
81 int pIdx = getPieceIndex(p->getSymbol(), p->getColor());
82
83 // Revert side
84 board.zobristKey ^= sideKey;
85

87 if (move.promotion != 0) {
88 int promoIdx = getPieceIndex(move.promotion, p->getColor());
89 board.zobristKey ^= pieceKeys[promoIdx][toSq];
90 }
91 else {
92 board.zobristKey ^= pieceKeys[pIdx][toSq];
93 }
94
95 // Revert captured
96 if (undo.pieceCaptured) {
97 int capIdx = getPieceIndex(undo.pieceCaptured->getSymbol(), undo.pieceCaptured->getColor());
98 board.zobristKey ^= pieceKeys[capIdx][toSq];

125{
126 undo.from = move.from;
127 undo.to = move.to;
128 undo.pieceMoved = board.getPieceAt(move.from);
129 undo.pieceCaptured = board.getPieceAt(move.to); // Might be nullptr
130 undo.promotion = move.promotion;
131
132 int fromSq = move.from.row * 8 + move.from.col;
133 int toSq = move.to.row * 8 + move.to.col;
134
135 Piece* p = board.getPieceAt(move.from);
136 int pIdx = getPieceIndex(p->getSymbol(), p->getColor());
137
138 // Zobrist remove piece
139 board.zobristKey ^= pieceKeys[pIdx][fromSq];
140 // Zobrist If captured, remove captured piece
141 if (undo.pieceCaptured) {
142 int capIdx = getPieceIndex(undo.pieceCaptured->getSymbol(), undo.pieceCaptured->getColor());
143 board.zobristKey ^= pieceKeys[capIdx][toSq];
144 }
145
146 // Zobrist insert piece
147 // If promoted put new figure
148 if (move.promotion != 0) {
149 int promoIdx = getPieceIndex(move.promotion, p->getColor());
150 board.zobristKey ^= pieceKeys[promoIdx][toSq];
151 }
152 else {
153 board.zobristKey ^= pieceKeys[pIdx][toSq];
154 }
155
156 // Switch side
157 board.zobristKey ^= sideKey;
158
159 // Raw move application
160 // Delete piece
161 board.squares[move.from.row][move.from.col] = nullptr;
162 // Place piece
163 board.squares[move.to.row][move.to.col] = undo.pieceMoved;
164
165 // Update piece position
166 if (undo.pieceMoved) {
167 undo.pieceMoved->setPosition(move.to);
168 }
169 // Handle promotion
170 if (move.promotion != 0) {
171 undo.pieceMoved->setSymbol(move.promotion);
172 }
173}

189 // If attacker is White (0), pawn must be 'below' (row-1) attacking up.

192
193 if (pawnRow >= 0 && pawnRow < 8) {
194 // Attack from left diagonal
195 if (pos.col - 1 >= 0) {
196 Piece* p = board.squares[pawnRow][pos.col - 1];
197 if (p && p->getColor() == attackerColor && toupper(p->getSymbol()) == 'P') return true;
198 }
199 // Attack from right diagonal
200 if (pos.col + 1 < 8) {
201 Piece* p = board.squares[pawnRow][pos.col + 1];
202 if (p && p->getColor() == attackerColor && toupper(p->getSymbol()) == 'P') return true;
203 }

375int eval(const Board& board, int color)
376{
377 int whiteScore = 0;
378 int blackScore = 0;
379
380 for (int row = 0; row < 8; ++row)
381 {
382 for (int col = 0; col < 8; ++col)
383 {
384 const Piece* cell = board.squares[row][col];
385 if (!cell || cell->symbol == 'E') continue;
386
387 int material = pieceValFromSymbol(cell->symbol);
388 // PST (Piece Square Table)
389 int positional = 0;
390 char sym = toupper(cell->symbol);
391 switch (sym) {
392 case 'P': positional = getPstValue(pawnPST, row, col, cell->color); break;
393 case 'N': positional = getPstValue(knightPST, row, col, cell->color); break;
394 case 'B': positional = getPstValue(bishopPST, row, col, cell->color); break;
395 case 'R': positional = getPstValue(rookPST, row, col, cell->color); break;
396 case 'Q': positional = getPstValue(queenPST, row, col, cell->color); break;
397 case 'K': positional = getPstValue(kingPST, row, col, cell->color); break;
398 }
399
400 int total = material + positional;
401 if (cell->color == 0) // White
402 whiteScore += total;
403 else // Black
404 blackScore += total;
405 }
406 }
407
408 int finalScore = whiteScore - blackScore;
409 return finalScore * color;
410}

462 if (move.pieceCaptured != nullptr) {
463 int victimVal = pieceValFromSymbol(move.pieceCaptured->getSymbol());
464 int attackerVal = pieceValFromSymbol(move.pieceMoved->getSymbol());
465
466 // Equation: Victim * 10 - Attacker.
467 return 10000 + (victimVal * 10) - attackerVal;
468 }
469
470 // PROMOTION BONUS
471 char sym = toupper(move.pieceMoved->getSymbol());
472 if (sym == 'P') {

476 return 9000;
477 }
478 }

480 return 0;

546int quiescence(Board& board, int alpha, int beta, int color)
547{
548 int stand = eval(board, color);
549 if (stand >= beta)
550 return beta;
551 if (stand > alpha)
552 alpha = stand;
553
554 auto caps = generateCaptures(board, to01(color));
555 // Only consider capture moves
556 orderMoves(caps);
557 for (auto& move : caps)
558 {
559 Undo undo;
560 applyMove(board, move, undo);
561
562 int score = -quiescence(board, -beta, -alpha, -color);
563 undoMove(board, move, undo);
564 if (score >= beta)
565 return beta;
566 if (score > alpha)
567 alpha = score;
568 }
569 return alpha;
570}

614int negamax(Board& board, int depth, int alpha, int beta, int color)
615{
616 if (isRepetition(board)) {
617 // 0 is equal position, slight minus for engine to avoid repetition
618 return -35;
619 }
620
621 ++nodesVisited;
622 int ttScore;
623 Move ttMove;
624 // Read hash
625 unsigned long long key = board.zobristKey;
626 if (TT.probe(key, depth, alpha, beta, ttScore, ttMove)) {
627 return ttScore;
628 // Score found in TT, immediately return
629 }
630
631 if (depth == 0 || gameOver(board))
632 return quiescence(board, alpha, beta, color);
633
634 auto moves = legalMoves(board, to01(color));
635
636 if (moves.empty()) {
637 // No legal moves, check for checkmate or stalemate
638 if (isInCheck(board, to01(color))) return -100000 + depth; // CHECKMATE
639 return 0; // STALEMATE
640 }
641
642 orderMoves(moves);
643
644 int best = -INF;
645 Move bestMove;
646 // For TT storage
647 int oldAlpha = alpha;
648 for (auto& move : moves)
649 {
650 Undo undo;
651 applyMove(board, move, undo);
652 int score = -negamax(board, depth - 1, -beta, -alpha, -color);
653 undoMove(board, move, undo);
654 if (score > best)
655 best = score;
656 if (score > alpha)
657 alpha = score;
658 if (alpha >= beta)
659 break;
660 // beta cutoff
661 }
662
663 TTFlag flag = TT_EXACT;
664 if (best <= oldAlpha) flag = TT_ALPHA; // Not better than alpha
665 else if (best >= beta) flag = TT_BETA;
666 // Cutoff
667
668 TT.store(key, best, depth, flag, bestMove);
669
670 return best;
671}
